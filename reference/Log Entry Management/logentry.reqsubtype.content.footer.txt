<script>
// Add event to [Submit] button to validate entry of subtype when page is submitted
function checkSubtype() {
    var selSubtype = $j('select[name="subtype"]');
    if(selSubtype[0].length > 1 && !selSubtype.val()) {
        psAlert({message:'Subtype must be entered.',title:'WARNING'});
        selSubtype.css("background-color","pink").focus();
									
        return false;
    }
    return true;
}
$j(document).ready(function() {
    $j('button[type="submit"]').click(checkSubtype);
});
</script>
